
Classe Arc :

    Attributs :
        Sommet : depart, arrivee
        Entier : poids
    
    Constructeur :

        Arc :
            Paramètres : Sommet depart, arrivee, Entier poids
            Code :
                depart de notre arc <- depart
                arrivee de notre arc <- arrivee
                poids de notre arc <- poids
            FinCode
        FinConstructeur

    Méthodes :

        Fonction getArrivee :
            Paramètres : aucun
            Code :
                Retourner arrivee de notre arc
            FinCode
        FinFonction

        Fonction getPoids :
            Paramètres : aucun
            Code :
                Retourner poids de notre arc
            FinCode
        FinFonction
